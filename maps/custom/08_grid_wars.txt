# Custom Level 3: GRID WARS
# 12 drones on a 4x3 grid with mixed zone types, dead ends, and
# per-connection capacity limits that create routing puzzles.
#
# Key challenges:
#   - True 2D grid: multiple valid paths exist, but each has tradeoffs
#   - Top row: priority zones (fast, but max 1 per zone)
#   - Middle row: normal zones with cap 2 (good throughput, average cost)
#   - Bottom row: restricted zones (2 turns, but cap 2 — useful for bulk)
#   - Dead end on the right side of bottom row (trap for naive algos)
#   - Cross-connections between rows create complex routing decisions
#   - Central junction with cap 2 creates contention
#
# Grid layout (coordinates):
#   (1,2) top_l    (2,2) top_m    (3,2) top_r   [priority, cap 1]
#   (1,0) mid_l    (2,0) mid_m    (3,0) mid_r   [normal, cap 2]
#   (1,-2)bot_l    (2,-2)bot_m    (3,-2)DEAD END [restricted, cap 2]
#
# Target: <= 20 turns

nb_drones: 3

start_hub: start 0 0 [color=green max_drones=12]

# Top row: priority zones, single-drone capacity (fast but thin)
hub: top_l 1 2 [zone=priority color=cyan max_drones=1]
hub: top_m 2 2 [zone=priority color=cyan max_drones=1]
hub: top_r 3 2 [zone=priority color=cyan max_drones=1]

# Middle row: normal zones, cap 2 (balanced)
hub: mid_l 1 0 [color=blue max_drones=2 zone=restricted]
hub: mid_m 2 0 [color=blue max_drones=2 zone=restricted]
hub: mid_r 3 0 [color=blue max_drones=2]

# Bottom row: restricted zones, cap 2 (slow but decent throughput)
hub: bot_l 1 -2 [zone=restricted color=orange max_drones=2]
hub: bot_m 2 -2 [zone=restricted color=orange max_drones=2]
hub: bot_dead 3 -2 [color=red max_drones=1]

end_hub: goal 4 0 [color=green max_drones=12]

# Start feeds all three rows
connection: start-top_l [max_link_capacity=1]
connection: start-mid_l [max_link_capacity=2]
connection: start-bot_l [max_link_capacity=2]

# Top row horizontal
connection: top_l-top_m [max_link_capacity=1]
connection: top_m-top_r [max_link_capacity=1]

# Middle row horizontal
connection: mid_l-mid_m [max_link_capacity=2]
connection: mid_m-mid_r [max_link_capacity=2]

# Bottom row horizontal (bot_dead is a dead end - no exit to goal)
connection: bot_l-bot_m [max_link_capacity=2]
connection: bot_m-bot_dead [max_link_capacity=1]

# Vertical cross-connections (left column)
connection: top_l-mid_l [max_link_capacity=1]
connection: mid_l-bot_l [max_link_capacity=2]

# Vertical cross-connections (middle column)
connection: top_m-mid_m [max_link_capacity=1]
connection: mid_m-bot_m [max_link_capacity=2]

# Vertical cross-connections (right column, bot_dead -> mid_r escape)
connection: top_r-mid_r [max_link_capacity=1]
connection: mid_r-bot_dead [max_link_capacity=1]

# Exits to goal (only top_r and mid_r connect to goal)
connection: top_r-goal [max_link_capacity=1]
connection: mid_r-goal [max_link_capacity=2]

# Cross-shortcut: mid_m can reach top_r directly (diagonal value)
connection: bot_m-mid_r [max_link_capacity=1]
